CREATE TABLE EXAM(
examNum INTEGER,
examName VARCHAR(30),
categoryID VARCHAR(3),
creatorEmpNum INTEGER,
dateCreated VARCHAR(35),
dateLastMod VARCHAR(35),
active NUMERIC(1),
displayAfterTaking NUMERIC(1));


CREATE TABLE EXAM_ENTRY(
examNum INTEGER,
examLoc INTEGER,
questionNum INTEGER,
entryTypeID VARCHAR(3));


CREATE TABLE ENTRY_TYPE(
entryTypeID VARCHAR(3),
entryType VARCHAR(20),
className VARCHAR(30));


CREATE TABLE SECTION_HEADER(
examNum INTEGER,
examLoc INTEGER,
header VARCHAR(75));


CREATE TABLE TRUE_FALSE(
examNum INTEGER,
examLoc INTEGER,
question VARCHAR(75),
solution NUMERIC(1));

CREATE TABLE ESSAY(
examNum INTEGER,
examLoc INTEGER,
question VARCHAR(75),
solution VARCHAR(200));

CREATE TABLE MATCHING(
examNum INTEGER,
examLoc INTEGER,
question VARCHAR(75));


CREATE TABLE WORD_BANK(
examNum INTEGER,
examLoc INTEGER,
question VARCHAR(75));


CREATE TABLE FILL_IN_BLANK(
examNum INTEGER,
examLoc INTEGER,
question VARCHAR(75));


CREATE TABLE MULT_CHOICE(
examNum INTEGER,
examLoc INTEGER,
question VARCHAR(75),
choice1 VARCHAR(75),
choice2 VARCHAR(75),
choice3 VARCHAR(75),
choice4 VARCHAR(75),
solution VARCHAR(75));


CREATE TABLE MATCHING_QUESTIONS(
examLoc INTEGER,
examNum INTEGER,
question VARCHAR(75),
choiceNum_of_solution INTEGER);


CREATE TABLE MATCHING_CHOICES(
examNum INTEGER,
examLoc INTEGER,
choiceNum INTEGER,
choice VARCHAR(75));


CREATE TABLE WORD_BANK_QUESTIONS(
examLoc INTEGER,
examNum INTEGER,
question VARCHAR(75),
choiceNum_of_solution INTEGER);


CREATE TABLE WORD_BANK_CHOICES(
examNum INTEGER,
examLoc INTEGER,
choiceNum INTEGER,
choice VARCHAR(75));


CREATE TABLE FILL_IN_BLANK_ANSWERS(
examNum INTEGER,
examLoc INTEGER,
answer VARCHAR(75));


CREATE TABLE EXAM_CATEGORY(
categoryID VARCHAR(5),
category VARCHAR(15));
